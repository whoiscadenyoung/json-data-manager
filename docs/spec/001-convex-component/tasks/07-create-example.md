### src/routes/schemas/index.tsx
```typescript
import { createFileRoute, Link } from "@tanstack/react-router";
import { useQuery } from "convex/react";
import { api } from "../../../convex/_generated/api";

export const Route = createFileRoute("/schemas/")({
  component: SchemasListPage,
});

function SchemasListPage() {
  const schemas = useQuery(api.schemas.list);

  return (
    <div className="p-8">
      <div className="flex justify-between items-center mb-6">
        <h1 className="text-2xl font-bold">JSON Schemas</h1>
        <Link
          to="/schemas/create"
          className="px-4 py-2 bg-blue-600 text-white rounded"
        >
          Create Schema
        </Link>
      </div>

      {schemas === undefined ? (
        <p>Loading...</p>
      ) : schemas.length === 0 ? (
        <p className="text-gray-500">No schemas yet.</p>
      ) : (
        <div className="grid gap-4">
          {schemas.map((schema) => (
            <Link
              key={schema._id}
              to="/schemas/$schemaId"
              params={{ schemaId: schema._id }}
              className="p-4 border rounded hover:bg-gray-50"
            >
              <h2 className="font-semibold">{schema.title}</h2>
              {schema.description && (
                <p className="text-sm text-gray-600">{schema.description}</p>
              )}
            </Link>
          ))}
        </div>
      )}
    </div>
  );
}
```

### src/routes/schemas/create.tsx
```typescript
import { createFileRoute, useNavigate } from "@tanstack/react-router";
import { SchemaEditor } from "@convex-dev/json-data-manager";

export const Route = createFileRoute("/schemas/create")({
  component: CreateSchemaPage,
});

function CreateSchemaPage() {
  const navigate = useNavigate();

  return (
    <div className="h-screen">
      <SchemaEditor
        mode="create"
        onSaved={(schemaId) => {
          navigate({ to: "/schemas/$schemaId", params: { schemaId } });
        }}
        onCancel={() => {
          navigate({ to: "/schemas" });
        }}
      />
    </div>
  );
}
```

### src/routes/schemas/$schemaId/index.tsx
```typescript
import { createFileRoute, Link } from "@tanstack/react-router";
import { useQuery } from "convex/react";
import { api } from "../../../../convex/_generated/api";
import { SchemaEditor } from "@convex-dev/json-data-manager";

export const Route = createFileRoute("/schemas/$schemaId/")({
  component: ViewSchemaPage,
});

function ViewSchemaPage() {
  const { schemaId } = Route.useParams();
  const schema = useQuery(api.schemas.get, { schemaId });
  const entryCount = useQuery(api.entries.countBySchema, { schemaId });

  if (schema === undefined) {
    return <div className="p-8">Loading...</div>;
  }

  if (schema === null) {
    return <div className="p-8">Schema not found</div>;
  }

  const hasEntries = (entryCount ?? 0) > 0;

  return (
    <div className="h-screen">
      <div className="p-4 border-b flex justify-between items-center">
        <div>
          <h1 className="text-xl font-bold">{schema.title}</h1>
          <p className="text-sm text-gray-600">
            {entryCount ?? 0} entries
          </p>
        </div>
        <div className="flex gap-2">
          <Link
            to="/schemas/$schemaId/data"
            params={{ schemaId }}
            className="px-4 py-2 bg-green-600 text-white rounded"
          >
            Manage Data
          </Link>
        </div>
      </div>

      <SchemaEditor
        mode={hasEntries ? "edit-with-entries" : "edit-no-entries"}
        schemaId={schemaId}
      />
    </div>
  );
}
```

  if (schema === undefined) {
    return <div className="p-8">Loading...</div>;
  }

    return <div className="p-8">Schema not found</div>;
  }

  return (
    <div className="h-screen p-8">
      <h1 className="text-2xl font-bold mb-6">
        Data: {schema.title}
      </h1>
      <DataManager schemaId={schemaId} />
    </div>
  );
}

# JSON Data Manager Example

This example demonstrates how to use the `@convex-dev/json-data-manager` component.

## Setup

1. Install dependencies:
   ```bash
   npm install
   ```

2. Set up Convex:
   ```bash
   npx convex dev
   ```

3. Start the dev server:
   ```bash
   npm run dev
   ```

## Features Demonstrated

- **Schema Management**: Create, edit, and delete JSON schemas
- **Visual Schema Builder**: Build schemas with a form-based UI
- **JSON Editor**: Direct JSON Schema editing with validation
- **Data Entry**: Auto-generated forms from schemas
- **Data Import**: Bulk import from JSON files
- **Real-time Updates**: All changes sync instantly via Convex

## Component API

### SchemaEditor

```tsx
import { SchemaEditor } from "@convex-dev/json-data-manager";

// Create mode
<SchemaEditor
  mode="create"
  onSaved={(schemaId) => console.log("Created:", schemaId)}
  onCancel={() => console.log("Cancelled")}
/>

// Edit without entries (full editing allowed)
<SchemaEditor
  mode="edit-no-entries"
  schemaId="xxx"
/>

// Edit with entries (restricted editing)
<SchemaEditor
  mode="edit-with-entries"
  schemaId="xxx"
/>
```

### DataManager

```tsx
import { DataManager } from "@convex-dev/json-data-manager";

<DataManager schemaId="xxx" />
