# Task 07: Create Example Application

## Objective
Create a complete example application that demonstrates how to consume the `@convex-dev/json-cms` component in a real project. The example is a full TanStack Start app that recreates the existing frontend functionality but uses the component.

import { createFileRoute, Link } from "@tanstack/react-router";
import { useQuery } from "convex/react";
import { api } from "../../convex/_generated/api";

export const Route = createFileRoute("/")({
  component: HomePage,
});

function HomePage() {
  const schemas = useQuery(api.schemas.list);

  return (
    <div className="p-8 max-w-4xl mx-auto">
      <div className="flex justify-between items-center mb-8">
        <h1 className="text-3xl font-bold">JSON CMS Example</h1>

          to="/schemas/create"
          className="px-4 py-2 bg-blue-600 text-white rounded hover:bg-blue-700"

          Create Schema
        </Link>
      </div>

      <p className="text-gray-600 mb-6">

      </p>

      <h2 className="text-xl font-semibold mb-4">Schemas</h2>

      {schemas === undefined ? (
        <p>Loading...</p>
      ) : schemas.length === 0 ? (
        <p className="text-gray-500">No schemas yet. Create one to get started.</p>
      ) : (
        <div className="grid gap-4">
          {schemas.map((schema) => (
            <Link
              key={schema._id}
              to="/schemas/$schemaId"
              params={{ schemaId: schema._id }}
              className="p-4 border rounded-lg hover:bg-gray-50 transition-colors"
            >
              <h3 className="font-semibold text-lg">{schema.title}</h3>
              {schema.description && (
                <p className="text-sm text-gray-600">{schema.description}</p>
              )}
            </Link>
          ))}
        </div>
      )}
    </div>
  );
}

import { createFileRoute, useNavigate } from "@tanstack/react-router";
import { SchemaEditor } from "@convex-dev/json-cms";

export const Route = createFileRoute("/schemas/create")({
  component: CreateSchemaPage,
});

function CreateSchemaPage() {
  const navigate = useNavigate();

  return (
    <div className="h-screen">

          navigate({ to: "/schemas" });

  );

### src/routes/schemas/$schemaId/index.tsx
```typescript
import { createFileRoute, Link } from "@tanstack/react-router";
import { useQuery } from "convex/react";
import { api } from "../../../../convex/_generated/api";
import { SchemaEditor } from "@convex-dev/json-cms";

export const Route = createFileRoute("/schemas/$schemaId/")({
  component: ViewSchemaPage,
});

function ViewSchemaPage() {
  const { schemaId } = Route.useParams();
  const schema = useQuery(api.schemas.get, { schemaId });
  const entryCount = useQuery(api.entries.countBySchema, { schemaId });

  if (schema === undefined) {
    return <div className="p-8">Loading...</div>;
  }

  if (schema === null) {
    return <div className="p-8">Schema not found</div>;
  }

  const hasEntries = (entryCount ?? 0) > 0;

  return (
    <div className="h-screen flex flex-col">
      <div className="p-4 border-b flex justify-between items-center bg-white">
        <div>
          <h1 className="text-xl font-bold">{schema.title}</h1>
          <p className="text-sm text-gray-600">{entryCount ?? 0} entries</p>
        </div>
        <div className="flex gap-2">
          <Link
            to="/schemas/$schemaId/create"
            params={{ schemaId }}
            className="px-4 py-2 bg-green-600 text-white rounded hover:bg-green-700"
          >
            Add Entry
          </Link>
        </div>
      </div>

      <div className="flex-1 overflow-hidden">
        <SchemaEditor
          mode={hasEntries ? "edit-with-entries" : "edit-no-entries"}
          schemaId={schemaId}
        />
      </div>
    </div>
  );
}
```

import { createFileRoute, useNavigate } from "@tanstack/react-router";
import { useQuery } from "convex/react";

import { SchemaEditor } from "@convex-dev/json-cms";

  component: EditSchemaPage,

  return (

        onSaved={() => {

        }}
        onCancel={() => {
          navigate({ to: "/schemas/$schemaId", params: { schemaId } });
        }}
      />
    </div>
  );
}

### README.md

# JSON CMS Example

This example demonstrates how to use the `@convex-dev/json-cms` component in a TanStack Start application.

## Setup

1. Install dependencies:
   ```bash
   bun install
   ```

2. Set up Convex:
   ```bash
   bunx convex dev
   ```

3. Start the dev server:
   ```bash
   bun run dev
   ```

## Features Demonstrated

- **Schema Management**: Create, edit, and delete JSON schemas
- **Visual Schema Builder**: Build schemas with a form-based UI
- **JSON Editor**: Direct JSON Schema editing with validation
- **Data Entry**: Auto-generated forms from schemas
- **Real-time Updates**: All changes sync instantly via Convex

## Component API

### SchemaEditor

```tsx
import { SchemaEditor } from "@convex-dev/json-cms";

// Create mode
<SchemaEditor
  mode="create"
  onSaved={(schemaId) => console.log("Created:", schemaId)}
  onCancel={() => console.log("Cancelled")}
/>

// Edit without entries (full editing allowed)
<SchemaEditor
  mode="edit-no-entries"
  schemaId="xxx"
/>

// Edit with entries (restricted editing)
<SchemaEditor
  mode="edit-with-entries"
  schemaId="xxx"
/>
```

### DataManager

import { DataManager } from "@convex-dev/json-cms";

<DataManager schemaId="xxx" />
```
